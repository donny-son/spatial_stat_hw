\section*{Q1}
\subsection*{(a)}
Prove $E(Y)=E(\mu+LZ)$ and $V(Y)=V(\mu+LZ)$

\begin{proof}
  \begin{align*}
    E(Y) &= E(\mu + LZ) \\
    &= \mu+E(LZ) \\
    &= \mu+LE(Z) & \text{(linearity property)} \\
    &= \mu \\
    V(Y) &= E([(Y-E(Y))(Y-E(Y))']) \\
    &= E([(Y-\mu)(Y-\mu)']) \\
    &= E([(LZ)(LZ)']) \\
    &= E([LZZ'L']) \\
    &= LE(ZZ')L' \\
    &= LV(Z)L' \\
    &= LIL' \\
    &= LL' \\
    &= \Sigma
  \end{align*}
\end{proof}
